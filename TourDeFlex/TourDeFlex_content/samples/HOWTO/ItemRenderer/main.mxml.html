<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml" backgroundGradientColors="[0x000000,0x323232]" 
    layout="absolute"
    applicationComplete="init()" viewSourceURL="srcview/index.html">

    <mx:Script>
        <![CDATA[

            import mx.collections.ArrayCollection;
            
            [Bindable]
            private var products:ArrayCollection;
            
            private function init():void
            {
                products = new ArrayCollection(model.product);
                dg.addEventListener("deleteRow", deleteRowHandler);
            }
                
            private function deleteRowHandler(event:Event):void
            {
                products.removeItemAt(dg.selectedIndex);
            }
            
        ]]>
    </mx:Script>

    <mx:Model id="model" source="catalog.xml"/>

    <mx:DataGrid id="dg" dataProvider="{products}" top="4" left="4" bottom="4" right="4" variableRowHeight="true">
        
        <mx:columns>

            <!-- You can define an inline component as the ItemRenderer for a DataGrid column -->
            <mx:DataGridColumn headerText="Name">
                <mx:itemRenderer>
                    <mx:Component>
                        <mx:Label text="{data.name}" fontWeight="bold" fontSize="12"/>
                    </mx:Component>
                </mx:itemRenderer>
            </mx:DataGridColumn>
            
            <!-- You can specify a class name as the ItemRenderer for a DataGrid column.
                We use "Text" here, but you could use Image, CheckBox, NumericStepper, etc. -->
            <mx:DataGridColumn dataField="description" itemRenderer="mx.controls.Text" headerText="Description"/>

            <!-- Another example of an inline component used as the ItemRenderer-->
            <mx:DataGridColumn>
                <mx:itemRenderer>
                    <mx:Component>
                        <mx:VBox horizontalScrollPolicy="off" verticalScrollPolicy="off">
                            <mx:CheckBox label="Triband" selected="{data.triband}"/>
                            <mx:CheckBox label="Camera" selected="{data.camera}"/>
                            <mx:CheckBox label="Video" selected="{data.video}"/>
                        </mx:VBox>
                    </mx:Component>
                </mx:itemRenderer>
            </mx:DataGridColumn>

            <!-- You can also specify a Custom Component as the ItemRenderer for a DataGrid column -->
            <mx:DataGridColumn itemRenderer="ImageRenderer" headerText="Image"/>

            <!-- Another example of a Custom Component. Notice that ColorRenderer is a very generic component and therefore 
                highly reusable: Unlike ImageRenderer, it is not specific to the data we are dealing with in this view. -->
            <mx:DataGridColumn dataField="monthToMonth" itemRenderer="ColorRenderer" textAlign="center" headerText="Month-to-Month Sales"/>
            <!-- Notice that we use the same ItemRenderer to display the month-by-month and the year-by-year sales. -->
            <mx:DataGridColumn dataField="yearToYear" itemRenderer="ColorRenderer" textAlign="center" headerText="Year-to-Year Sales"/>
            
            <!-- Another example of an inline component used as the ItemRenderer-->
            <mx:DataGridColumn width="40">
                <mx:itemRenderer>
                    <mx:Component>
                        <!-- The second parameter of the Event constructor (true) indicates that the event should "bubble".
                            This allows us to listen for the deleteRow event on the DataGrid (see init function) --> 
                        <mx:Button  icon="@Embed('icon_delete.png')" click="dispatchEvent(new Event('deleteRow', true))" skin="{null}"/>
                    </mx:Component>
                </mx:itemRenderer>
            </mx:DataGridColumn>
            
        </mx:columns>
        
    </mx:DataGrid>
    
</mx:Application>
