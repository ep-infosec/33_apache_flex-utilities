<?xml version="1.0" encoding="utf-8"?>
<mx:Canvas xmlns:mx="http://www.adobe.com/2006/mxml" borderStyle="outset" backgroundColor="#333333">

    <mx:Metadata>
        [Event(name="complete", type="mx.events.VideoEvent")]
        [Event(name="stateChange", type="mx.events.VideoEvent")]
    </mx:Metadata>
    
    <mx:Script>
        <![CDATA[
            import mx.events.VideoEvent;
            import mx.events.ItemClickEvent;
            
            [Bindable] public var videoWidth:int = 240;
            [Bindable] public var videoHeight:int = 180;
            [Bindable] public var source:String; 

            private function buttonBarClick(event:ItemClickEvent):void
            {
                switch (event.index)
                {
                    case 0:
                        videoDisplay.play();
                        break;
                    case 1:
                        videoDisplay.pause();
                        break;
                    case 2:
                        videoDisplay.stop();
                        break;
                }
            }
            
            private function playHeadUpdateHandler(event:VideoEvent):void
            {
                progressBar.setProgress(event.playheadTime, videoDisplay.totalTime);
            }
            
        ]]>
    </mx:Script>
    
    <mx:VideoDisplay id="videoDisplay" width="{videoWidth}" height="{videoHeight}" autoPlay="false"
        top="4" left="4" right="4"
        playheadUpdate="playHeadUpdateHandler(event)"
        source="{source}"
        complete="dispatchEvent(event)"
        stateChange="dispatchEvent(event)"/>
    
    <mx:ProgressBar id="progressBar" mode="manual" minimum="0" maximum="{videoDisplay.totalTime}" 
        label="" top="{videoHeight + 8 }" left="4" right="4"/>
    
    <mx:ButtonBar id="bb" itemClick="buttonBarClick(event)" top="{videoHeight + 20}" bottom="4" horizontalCenter="0" toolTipField="toolTip">
        <mx:dataProvider>
            <mx:Object icon="@Embed('assets/control_play_blue.png')" toolTip="Play"/>
            <mx:Object icon="@Embed('assets/control_pause_blue.png')" toolTip="Pause"/>
            <mx:Object icon="@Embed('assets/control_stop_blue.png')" toolTip="Stop"/>
        </mx:dataProvider>
    </mx:ButtonBar>

</mx:Canvas>
