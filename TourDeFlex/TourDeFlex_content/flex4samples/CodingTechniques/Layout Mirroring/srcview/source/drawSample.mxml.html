<s:Application xmlns:fx="http://ns.adobe.com/mxml/2009" 
               xmlns:s="library://ns.adobe.com/flex/spark" xmlns:mx="library://ns.adobe.com/flex/mx"
               creationComplete="application1_creationCompleteHandler(event)">
    <!-- NOTE: Requires Flex 4.1 SDK - download here: http://opensource.adobe.com/wiki/display/flexsdk/Download+Flex+4 -->
    <fx:Script>
        <![CDATA[
            import mx.core.UIComponent;
            import mx.events.FlexEvent;

            private var myShape:Shape = new Shape();
            private var checked:Boolean = false;
            
            protected function application1_creationCompleteHandler(event:FlexEvent):void
            {

                myPanel.addEventListener(MouseEvent.MOUSE_DOWN, onDown);
                myPanel.addEventListener(MouseEvent.MOUSE_MOVE, onMove);
                myPanel.addEventListener(MouseEvent.MOUSE_UP, onUp);
                
                var UIRef:UIComponent = new UIComponent();
                myPanel.addElement(UIRef);
                UIRef.addChild(myShape);
            }
            
            protected function onDown(evt:MouseEvent):void{
                checked = true;
                myShape.graphics.lineStyle(1,0xFF0000,1);
                myShape.graphics.moveTo(evt.stageX,evt.stageY - 55);
            }
            
            protected function onMove(evt:MouseEvent):void{
                if(checked){
                    myShape.graphics.lineTo(evt.stageX,evt.stageY - 55);
                }
            }
            
            protected function onUp(evt:MouseEvent):void{
                
                checked = false;
            }
            
            protected function togglebutton1_clickHandler(event:MouseEvent):void
            {
    
                if(myCB.selected){
                    myPanel.layoutDirection = "rtl";
                }else{
                    myPanel.layoutDirection = "ltr";
                }
            }
            


        ]]>
    </fx:Script>
    <s:Panel title="Layout Mirroring Draw Sample" skinClass="TDFPanelSkin" width="100%" height="100%">
        <s:layout>
            <s:VerticalLayout gap="8" horizontalAlign="center"/>
        </s:layout>
        <s:CheckBox id="myCB" horizontalCenter="0" verticalCenter="0" y="4" label="Right to Left" selected="false" click="togglebutton1_clickHandler(event)"/>
        <s:Panel y="30" width="446" height="170" title="Coordinates Sample" id="myPanel" horizontalCenter="0">
            <s:Label horizontalCenter="0" verticalCenter="0" text="Draw something with your mouse" width="424" verticalAlign="middle" textAlign="center" color="#737373"/>
        </s:Panel>
    </s:Panel>
    
</s:Application>
