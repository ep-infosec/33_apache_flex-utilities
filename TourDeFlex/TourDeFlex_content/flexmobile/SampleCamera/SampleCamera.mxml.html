<?xml version="1.0" encoding="utf-8"?>
<s:View xmlns:fx="http://ns.adobe.com/mxml/2009" xmlns:mx="library://ns.adobe.com/flex/mx" 
               xmlns:s="library://ns.adobe.com/flex/spark" title="SampleCamera" 
	       viewActivate="view1_viewActivateHandler(event)" removing="view1_removingHandler(event)">
    
	<fx:Script>
        <![CDATA[
            import mx.events.FlexEvent;
            
            protected var v:Video;
            protected var c:Camera;
            
            protected function view1_viewActivateHandler(event:FlexEvent):void
            {
                if (Camera.isSupported)
                {
                    v = new Video(width, height);
                    uic.addChild(v);
                    c = Camera.getCamera();
                    c.setMode(width, height, 10);
                    v.attachCamera(c);
                    
                }
                else lbl.text = "Camera not supported";
            }
            
            
            protected function view1_removingHandler(event:FlexEvent):void
            {
                v = null;
                c = null;
            }
            
        ]]>
    </fx:Script>
    <s:Label id="lbl"/>
    <mx:UIComponent id="uic" width="100%" height="100%"/>
</s:View>
