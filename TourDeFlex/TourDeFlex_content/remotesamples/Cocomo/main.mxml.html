        <![CDATA[
        
            import com.adobe.rtc.events.AuthenticationEvent;
            import com.adobe.rtc.events.ConnectSessionEvent;
            import com.adobe.rtc.events.SessionEvent;
            
            private function init():void
            {
                session.addEventListener(SessionEvent.ERROR, cocomoEventHandler);
                session.addEventListener(SessionEvent.SYNCHRONIZATION_CHANGE, cocomoEventHandler);
                auth.addEventListener(AuthenticationEvent.AUTHENTICATION_FAILURE, cocomoEventHandler);
                auth.addEventListener(AuthenticationEvent.AUTHENTICATION_SUCCESS, cocomoEventHandler);
            }
            
            private function cocomoEventHandler(event:Event):void
            {
                switch (event.type)
                {
                    case SessionEvent.SYNCHRONIZATION_CHANGE:    
                        if (session.isSynchronized) {
                            //  We are connected to the room
                            statusMessage.text = "Connected to room " + session.roomURL;
                            session.roomManager.autoPromote = true;
                            vs.selectedIndex = 1;
                        } 
                        else 
                        {
                            // We are disconnected from the room
                            session.roomURL = null;
                            notificationMessage.text = "Disconnected";
                            vs.selectedIndex = 0;
                        }
                        break;                    

                    case AuthenticationEvent.AUTHENTICATION_SUCCESS:    
                        notificationMessage.text = "Authentication Succeeded";
                        break;                    

                    case AuthenticationEvent.AUTHENTICATION_FAILURE:    
                        notificationMessage.text = "Authentication Failed";
                        break;                    

                    case SessionEvent.ERROR:    
                        var sError:SessionEvent = event as SessionEvent;
                        notificationMessage.text = sError.error.message;
                        break;                    

                    default:
                        notificationMessage.text = "Got event " + event;
                }
            }
            
            private function login():void
            {
                notificationMessage.text = "";
                auth.userName = username.text;
                auth.password = pwd.visible ? password.text : null; // password==null : the user is a guest
                session.roomURL = roomURL.text;        
                session.login();
            }
        ]]>
    </mx:Script>
