    <mx:Script>
        <![CDATA[
            import mx.events.AIREvent;
            public var _lastTime: Date = new Date;                // last entry into the EnterFrame handler
            public var _targetFPS: Number = 24;                    // requested framerate
            
            // Ease function to make the logo blink slowly
            public function ease(t:Number): Number {
                return (Math.sin(Math.PI * (2 * t - 0.5)) + 1 ) / 2;
            }
            
            
            // This handler is called anytime a new frame is ready to be drawn
            // Typically, it's going to be called as frequently as dictated by the 
            // frameRate of the application
            public function enterFrame(event:Event):void {
                var currTime : Date = new Date();
                if (currTime.time - _lastTime.time > 10)
                {
                       fps.text = Math.round(1000 / (currTime.time - _lastTime.time)).toString();
                       _lastTime = currTime;
                }
                
                   logo.alpha = 0.2 + ease(currTime.time / 3000);
            } 

            public function initApp():void 
            {
                // Request notification whenever the stage is drawn
                this.addEventListener(Event.ENTER_FRAME, enterFrame);

                // Request notification whenever the app is activated or deactivated
                this.addEventListener(AIREvent.APPLICATION_ACTIVATE, applicationActivate);
                this.addEventListener(AIREvent.APPLICATION_DEACTIVATE, applicationDeactivate);
            }
 
             // The application is being brough to the foreground
            public function applicationActivate(event:Event):void {
                stage.frameRate = _targetFPS;
                targetFPS.text = _targetFPS.toString() + " fps";
                this.addEventListener(Event.ENTER_FRAME, enterFrame);
            } 


             // The application is being sent to the background
            public function applicationDeactivate(event:Event):void {
                this.removeEventListener(Event.ENTER_FRAME, enterFrame);
                stage.frameRate = 0.1;
                targetFPS.text = "--";
                fps.text = "--";
            } 


            public function changeFrameRate(delta:Number):void 
            {
                stage.frameRate = stage.frameRate + delta;
                _targetFPS = stage.frameRate;
                targetFPS.text = Math.round(stage.frameRate).toString() + " fps";
            }

 
        ]]>
