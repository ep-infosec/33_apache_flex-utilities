<?xml version="1.0" encoding="utf-8"?>
<mx:Application xmlns:mx="http://www.adobe.com/2006/mxml"
    backgroundGradientColors="[0x000000,0x323232]"
    paddingTop="8" paddingBottom="8" paddingLeft="8" paddingRight="8" viewSourceURL="srcview/index.html">

    <mx:Script>
        <![CDATA[

            import mx.controls.dataGridClasses.DataGridColumn;
            import mx.rpc.events.ResultEvent;

            private function resultHandler(event:ResultEvent):void
            {
                dg.dataProvider = event.result.ItemSearchResponse.Items.Item;
            }
            
            private function getAuthorLabel(item:Object, column:DataGridColumn):String
            {
                if (item==null || item.ItemAttributes == null || item.ItemAttributes.Author == null) return "";
                return item.ItemAttributes.Author.toString();
            }

            private function getTitleLabel(item:Object, column:DataGridColumn):String
            {
                if (item==null || item.ItemAttributes == null || item.ItemAttributes.Title == null) return "";
                return item.ItemAttributes.Title;
            }

            private function getPriceLabel(item:Object, column:DataGridColumn):String
            {
                if (item==null || item.ItemAttributes == null || item.ItemAttributes.ListPrice == null) return "";
                return item.ItemAttributes.ListPrice.FormattedPrice;
            }
            
        ]]>
    </mx:Script>

    <mx:HBox>
        <mx:Label text="Keywords:" color="#FFFFFF"/>
        <mx:TextInput id="searchStr" text="Adobe Flex"/>
        <mx:Button label="Search Books on Amazon" click="srv.send()"/>
    </mx:HBox>

    <mx:HTTPService id="srv"
        showBusyCursor="true"
        url="http://ecs.amazonaws.com/onca/xml" method="POST"
        result="resultHandler(event)">
        <mx:request >
            <Version>2008-08-19</Version>
            <Service>AWSECommerceService</Service>
            <Operation>ItemSearch</Operation>
            <AWSAccessKeyId>[YOUR KEY HERE]</AWSAccessKeyId>
            <Keywords>{escape(searchStr.text)}</Keywords>
            <SearchIndex>Books</SearchIndex>
            <ResponseGroup>Medium,Images</ResponseGroup>
        </mx:request> 
    </mx:HTTPService>
    
    <mx:DataGrid id="dg" width="100%" height="100%" rowHeight="160" minHeight="200" minWidth="200">
        <mx:columns>
            <mx:DataGridColumn>
                <mx:itemRenderer>
                    <mx:Component>
                        <mx:Image source="{data.MediumImage.URL}"/>
                    </mx:Component>
                </mx:itemRenderer>
            </mx:DataGridColumn>
            <mx:DataGridColumn labelFunction="getTitleLabel" headerText="Title" itemRenderer="mx.controls.Text"/>
            <mx:DataGridColumn labelFunction="getAuthorLabel" headerText="Authors" itemRenderer="mx.controls.Text"/>
            <mx:DataGridColumn labelFunction="getPriceLabel" headerText="Price" textAlign="right"/>
        </mx:columns>
    </mx:DataGrid>
    
</mx:Application>
