    <fx:Script>
        <![CDATA[
            import flash.events.AccelerometerEvent;
            import flash.sensors.Accelerometer;
            import mx.controls.Alert;
            
            private var acc:Accelerometer = new Accelerometer();
            
            private function sense():void
            {
                Alert.show("Sensing - is supported value is " + Accelerometer.isSupported);
                acc.addEventListener(AccelerometerEvent.UPDATE,onUpdate);
                acc.setRequestedUpdateInterval(60);
            }
            
            private function onUpdate(e:AccelerometerEvent):void
            {
                Alert.show("Triggered Update! Acceleration X is: " + e.accelerationX + " Y " + e.accelerationY + " z " + e.accelerationZ);
                trace("Triggered");
            }
            
        ]]>
    </fx:Script>
    <s:Button horizontalCenter="0" verticalCenter="0" label="Go" click="sense()"/>
</s:Application>
