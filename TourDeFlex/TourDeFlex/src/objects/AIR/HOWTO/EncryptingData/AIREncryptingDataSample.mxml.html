<s:WindowedApplication xmlns:fx="http://ns.adobe.com/mxml/2009" 
                       xmlns:s="library://ns.adobe.com/flex/spark" 
                       xmlns:mx="library://ns.adobe.com/flex/mx" 
                       backgroundColor="#323232" styleName="plain" width="100%" height="100%" viewSourceURL="srcview/index.html">

<fx:Script>
    <![CDATA[
        
        import mx.controls.Alert;
        
        private function storeLoginData():void
        {
            if (userid.text!=null) 
            {
                var pwBytes:ByteArray = new ByteArray();
                pwBytes = new ByteArray();
                pwBytes.writeUTFBytes(password.text);
                EncryptedLocalStore.setItem(userid.text, pwBytes);
                Alert.show("Login successful! Userid/password stored for future retrieval.");
             }
             else Alert.show("No userid entered.");
            
        }
        private function login():void
        {
            if (chkRemember.selected) {
                storeLoginData();
            }
            else Alert.show("Login successful! Userid/password not stored.");
            clear();

        }
        private function retrievePassword():void
        {
            if (userid.text!=null && userid.text.length>0) {
                var storedValue:ByteArray = EncryptedLocalStore.getItem(userid.text);
                if (storedValue!=null) {
                    password.text = storedValue.readUTFBytes(storedValue.length)
                }
                else Alert.show("No stored data found for userid: " + userid.text);
            }
            else Alert.show("Please enter a userid to retrieve the password for. " + userid.text);
        }
        private function clear():void
        {
            userid.text="";
            password.text="";
            chkRemember.selected=false;    
        }
        private function removeLoginData():void
        {
            if (userid.text != null && userid.text.length>0) {
                EncryptedLocalStore.removeItem(userid.text);
                Alert.show("Login data removed for userid: " + userid.text);
            }
            else Alert.show("Please enter the userid of which to remove the stored password.");
        }
    ]]>
</fx:Script>

    <s:VGroup paddingLeft="70" paddingTop="80">
        <mx:Form>
            <mx:FormItem label="Userid:" color="white">
                <mx:TextInput id="userid" color="black"/>
            </mx:FormItem>
            <mx:FormItem label="Password:" color="white">
                <mx:TextInput id="password" color="black" displayAsPassword="true"/>
            </mx:FormItem>
            <mx:FormItem label="Remember Password:" color="white">
                <mx:CheckBox id="chkRemember" />
            </mx:FormItem>
            <mx:FormItem direction="horizontal">
                <s:Button left="165" top="110" label="Login" click="login()"/>
                <s:Button left="330" top="110" label="Retrieve Password" click="retrievePassword()"/>
            </mx:FormItem>
        </mx:Form>    
        <s:Button label="Remove Data Store" click="removeLoginData()" width="100%"/>    
    </s:VGroup>
</s:WindowedApplication>
