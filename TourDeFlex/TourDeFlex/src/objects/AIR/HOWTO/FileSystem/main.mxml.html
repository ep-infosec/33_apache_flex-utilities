<?xml version="1.0" encoding="utf-8"?>
<s:WindowedApplication xmlns:fx="http://ns.adobe.com/mxml/2009" 
                       xmlns:s="library://ns.adobe.com/flex/spark" 
                       xmlns:mx="library://ns.adobe.com/flex/mx" 
                       backgroundColor="#323232" viewSourceURL="srcview/index.html">
    
    <fx:Script>
        private const textFilter:FileFilter = new FileFilter("Text Files (txt, html, htm)","*.txt;*.html;*.htm;");
    </fx:Script>
    
    <s:VGroup horizontalAlign="center" horizontalCenter="0" paddingTop="5">
        <mx:RichTextEditor id="rte" width="100%" height="220" title="Text Editor"/>
        <s:HGroup>
            <s:Button label="Save">
                <s:click>
                    var f:File = File.desktopDirectory;
                    f.browseForSave("Save As");
                    f.addEventListener(Event.SELECT, function (event:Event):void {
                    var stream:FileStream = new FileStream();
                    stream.open((event.target as File),FileMode.WRITE);
                    stream.writeUTFBytes(rte.htmlText);
                    stream.close();
                    });
                </s:click>
            </s:Button>
            
            <s:Button label="Open">
                <s:click>
                    var f:File = File.desktopDirectory;
                    f.browseForOpen("Select file to open", [textFilter]);
                    f.addEventListener(Event.SELECT, function (event:Event):void {
                    var fs:FileStream = new FileStream();
                    fs.open(event.target as File, FileMode.READ);
                    rte.htmlText = fs.readUTFBytes(fs.bytesAvailable);
                    fs.close();
                    });
                </s:click>
            </s:Button>        
        </s:HGroup>      
    </s:VGroup>
     
</s:WindowedApplication>
