<?xml version="1.0" encoding="utf-8"?>
<mx:Module xmlns:fx="http://ns.adobe.com/mxml/2009" xmlns:s="library://ns.adobe.com/flex/spark" 
                       xmlns:mx="library://ns.adobe.com/flex/mx" backgroundColor="0x323232" 
                       height="100%" width="100%" remove="toastWindow.close()">

    <fx:Declarations>
        <mx:Move id="myMov"/>        
    </fx:Declarations>
    
    <fx:Script>
        <![CDATA[
            import flash.display.Screen;
            private var toastWindow:MovieWindow = new MovieWindow();
        
            private function showToast():void {
                var mainScreen:Screen = Screen.mainScreen;
                toastWindow.open();
                myMov.target = toastWindow;
                myMov.xTo = myMov.xFrom = mainScreen.visibleBounds.width - toastWindow.width - 50;
                myMov.yFrom = mainScreen.visibleBounds.height;
                myMov.yTo = mainScreen.visibleBounds.height - toastWindow.height + 50; 
                myMov.duration = 1000;
                myMov.play();
            }
        ]]>
    </fx:Script>
    
    <s:HGroup horizontalCenter="0" verticalCenter="0" x="247" y="24">
        <s:Button label="Lee Brimelow Video Reminder"  click="showToast()" horizontalCenter="0" verticalCenter="0"/>
    </s:HGroup>
    
</mx:Module>
