            private function init():void {
                // Retrieve data frmo the app descriptor

                var appId:String = appXml.ns::id[0]; 
                var appVersion:String = appXml.ns::version[0]; 
                var appName:String = appXml.ns::filename[0]; 

                message.text = "Application Descriptor Data:\n\n";
                message.text += "   Application ID: " + appId + "\n"; 
                message.text += "   Version: " + appVersion + "\n"; 
                message.text += "   Filename: " + appName + "\n\n";
                message.text += "   Publisher ID: " + NativeApplication.nativeApplication.publisherID + "\n"; 
            }
